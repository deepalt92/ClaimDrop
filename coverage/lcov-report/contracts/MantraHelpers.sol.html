        if (v == 0) {
            return "0";

        uint256 tmp = v;

            digits++;

    function pad6(uint256 v) internal pure returns (string memory) {
        // returns 6-digit zero-padded string for v < 1e6
        bytes memory b = new bytes(6);

            v /= 10;
        }
        return string(b);
    }
